4 . Leap year and palindrome

leap_yr.c

#include <stdio.h> void main() { int yr; printf("Enter year: "); scanf("%d",&yr); if(yr%4 == 0) { if(yr%100 == 0) { if (yr%400 == 0) printf("%d is a Leap Year", yr); else printf("%d is not a Leap Year", yr); } else printf("%d is a Leap Year", yr); } else printf("%d is not a Leap Year", yr); }

palindrome.c

#include <stdio.h> void main() { int num, reverse_num=0, rem,temp; printf("Enter an integer: "); scanf("%d", &num); temp=num; while(temp!=0) { rem=temp%10; reverse_num=reverse_num*10+rem; temp/=10; } if(reverse_num==num) printf("%d is a palindrome number",num); else printf("%d is not a palindrome number",num); }

5 . Size of and swap

size_of.c

#include <stdio.h> void main() { printf("Size of char: %ld byte\n",sizeof(char)); printf("Size of int: %ld bytes\n",sizeof(int)); printf("Size of float: %ld bytes\n",sizeof(float)); printf("Size of double: %ld bytes", sizeof(double)); }

swap.c

#include <stdio.h> void main() { int a,b,c; printf("Enter 2 nos.: "); scanf("%d%d",&a,&b); printf("BEFORE SWAPPING\n"); printf("a = %d\n",a); printf("b = %d\n",b); c=a; a=b; b=c; printf("AFTER SWAPPING\n"); printf("a = %d\n",a); printf("b = %d\n",b); }
